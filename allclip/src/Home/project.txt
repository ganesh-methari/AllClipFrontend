# ЁЯОм AllClip Downloader (Full Beginner-Friendly Version)

A YouTube downloader web app built with **React + Node + Express + yt-dlp + FFmpeg**.  
Features:

- Download full audio/video from YouTube
- Preview video thumbnail before download
- Clip video/audio by specifying start & end time
- Free-friendly deployment (Render + Vercel/Netlify)

тЪЩя╕П Backend Setup (Node + Express)
Navigate to backend:
cd backend
npm install express cors body-parser
Example server.js:
const express = require("express");
const cors = require("cors");
const { exec } = require("child_process");
const bodyParser = require("body-parser");
const path = require("path");

const app = express();
app.use(cors());
app.use(bodyParser.json());

const DOWNLOAD_DIR = path.join(__dirname, "downloads");

app.post("/download", (req, res) => {
    const { url, format = "mp3", start, end } = req.body;

    if (!url) return res.status(400).json({ error: "URL required" });

    // Command for yt-dlp + FFmpeg clipping
    let cmd = `yt-dlp -x --audio-format ${format} -o "${DOWNLOAD_DIR}/%(title)s.%(ext)s" "${url}"`;
    if (start || end) {
        cmd += ` --external-downloader ffmpeg --external-downloader-args "-ss ${start || 0} ${end ? `-to ${end}` : ""}"`;
    }

    console.log("Running:", cmd);

    exec(cmd, (err, stdout, stderr) => {
        if (err) {
            console.error(stderr);
            return res.status(500).json({ error: stderr });
        }
        console.log(stdout);
        res.json({ message: "Download complete!" });
    });
});

const PORT = 3000;
app.listen(PORT, () => console.log(`Server running on http://localhost:${PORT}`));

Example Hero.jsx with thumbnail + clip timing:
import React, { useState } from "react";
import axios from "axios";

export default function Hero() {
  const [url, setUrl] = useState("");
  const [thumb, setThumb] = useState("");
  const [start, setStart] = useState("");
  const [end, setEnd] = useState("");
  const [message, setMessage] = useState("");

  const fetchThumbnail = async () => {
    // Extract thumbnail using yt-dlp json
    try {
      const res = await axios.post("http://localhost:3000/download", { url, format: "json" });
      setThumb(res.data.thumbnail);
    } catch (err) {
      console.error(err);
    }
  };

  const handleDownload = async () => {
    try {
      const res = await axios.post("http://localhost:3000/download", { url, format: "mp3", start, end });
      setMessage(res.data.message);
    } catch (err) {
      console.error(err);
      setMessage("Download failed");
    }
  };

  return (
    <div>
      <h2>AllClip Downloader</h2>
      <input placeholder="Paste YouTube URL" value={url} onChange={e => setUrl(e.target.value)} />
      <button onClick={fetchThumbnail}>Show Thumbnail</button>
      {thumb && <img src={thumb} alt="Thumbnail" style={{ width: 200, margin: 10 }} />}
      <div>
        <input placeholder="Start (sec)" value={start} onChange={e => setStart(e.target.value)} />
        <input placeholder="End (sec)" value={end} onChange={e => setEnd(e.target.value)} />
      </div>
      <button onClick={handleDownload}>Download Audio</button>
      <p>{message}</p>
    </div>
  );
}
